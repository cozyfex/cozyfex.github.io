appVersion: apps/v1
kind: ReplicaSet
metadata:
  name: simple-webapp
  labels:
    app: App1
    function: Front-end
spec:
  replicas: 3
  selector: # -> This selector is matching below labels in template.
    matchLabels:
      app: App1
  template:
    metadata:
      labels:
        app: App1
        function: Front-end
    spec:
      containers:
        - name: simple-webapp
          image: simple-webapp

  name: simple-webapp
  labels:
    app: App1
    function: Front-end
  annotations:
    buildVersion: 1.34
spec:
  replicas: 3
  selector: # -> This selector is matching below labels in template.
    matchLabels:
      app: App1
  template:
    metadata:
      labels:
        app: App1
        function: Front-end
    spec:
      containers:
