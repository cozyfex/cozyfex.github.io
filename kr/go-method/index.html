package main

type Rect struct {
	width, height int
}

// Value receiver
func (r Rect) areaWithValue() int {
	r.width++
	return r.width * r.height
}

// Pointer receiver
func (r *Rect) areaWithPointer() int {
	r.width++
	return r.width * r.height
}

func main() {
	rect := Rect{10, 20}
	println(rect.areaWithValue()) // 220
	println(rect.width) // 10
	println(rect.areaWithPointer()) // 220
	println(rect.width) // 11
}
